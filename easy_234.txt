bool isPalindrome(ListNode* head) {
        if (head == NULL)
            return true;
        ListNode *slow = head, *fast = head;
        stack<int> st;
        while (fast != NULL && fast->next != NULL) {
            st.push(slow->val);
            slow = slow->next;
            fast = fast->next->next;
        }
        if (fast != NULL)
            slow = slow->next;
        while (slow != NULL) {
            if (slow->val == st.top()){
                st.pop();
                slow=slow->next;
            }
            else
                return false;
        }
        return true;
}
